je_yue_dict = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"
	group = je_group_historical_content

	immediate = {
		random_interest_group = {
			limit
			{
				is_interest_group_type = ig_armed_forces
				exist = leader
			}
			leader = {
				save_scope_as = armed_forces_leader
			}
		}
		random_interest_group = {
			limit
			{
				is_interest_group_type = ig_landowners
				exist = leader
			}
			leader = {
				save_scope_as = landowners_leader
			}
		}
		random_interest_group = {
			limit
			{
				is_interest_group_type = ig_petty_bourgeoisie
				exist = leader
			}
			leader = {
				save_scope_as = petty_leader
			}
		}
	}

	complete = {
		has_law = law_type:law_monarchy
		OR = {
			has_law = law_type:law_autocracy
			has_law = law_type:law_oligarchy
		}
	}
	on_complete = {
		trigger_event = {
			id = papal_to_rome.6
		}
		remove_modifier = modifier_yua_base_two
		remove_modifier = diplomatic_plays_blocked
	}
	fail = {
		scope:armed_forces_leader = {
			NOT = {
				has_ideology = ideology:ideology_authoritarian
			}
		}
		scope:landowners_leader = {
			NOT = {
				has_ideology = ideology:ideology_authoritarian
			}
		}
		scope:petty_leader = {
			NOT = {
				has_ideology = ideology:ideology_authoritarian
			}
		}
	}
	on_fail = {
		trigger_event = {
			id = papal_to_rome.7
		}
	}
	should_be_pinned_by_default = yes
}
