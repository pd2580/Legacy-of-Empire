je_ye_button_one = {
	name = "je_ye_button_one"
	desc = "je_ye_button_one_desc"
	visible = {
		always = yes
	}
	ai_chance = {
		base = 100
	}
	possible = {
		c:SAO = {
			relations:root >= relations_threshold:amicable
		}
	}
	effect = {
		random_state = {
			limit = {
				state_region = s:STATE_OREGON
			}
			set_state_owner = ROOT
		}
		random_state = {
			limit = {
				state_region = s:STATE_IDAHO
			}
			set_state_owner = ROOT
		}
		random_state = {
			limit = {
				state_region = s:STATE_WASHINGTON
			}
			set_state_owner = ROOT
		}
		add_modifier = {
			name = modifier_yua_one
			months = 60
		}
		hidden_effect = {
			s:STATE_OREGON = {
				remove_homeland = cu:salish
				remove_homeland = cu:nez_perce
				add_homeland = cu:han
				add_homeland = cu:yue
				add_homeland = cu:min
				add_homeland = cu:manchu
			}
			s:STATE_IDAHO = {
				remove_homeland = cu:salish
				remove_homeland = cu:nez_perce
				remove_homeland = cu:paiute
				add_homeland = cu:han
				add_homeland = cu:yue
				add_homeland = cu:min
				add_homeland = cu:manchu
			}
			s:STATE_WASHINGTON = {
				remove_homeland = cu:salish
				remove_homeland = cu:nez_perce
				add_homeland = cu:han
				add_homeland = cu:yue
				add_homeland = cu:min
				add_homeland = cu:manchu
			}
		}
	}
}
