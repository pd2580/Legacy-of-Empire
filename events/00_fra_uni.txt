fra_uni.1 = {
	type = country_event
	placement = root

	title = fra_uni.1.t
	desc = fra_uni.1.d
	flavor = fra_uni.1.f

	event_image = {
		video = "europenorthamerica_capitalists_meeting"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	duration = 3

	trigger = {

	}

	immediate = {

	}

	option = {
		name = fra_uni.1.a
		default_option = yes
		add_journal_entry = {
			type = je_francia_cent_bgy
		}
	}
	option = {
		name = fra_uni.1.b
		set_variable = {
			name = fra_uni_var
			value = 0
		}
		add_journal_entry = {
			type = je_francia_reuni_bgy
		}
		add_journal_entry = {
			type = je_francia_reuni_bgy_fra
		}
		add_journal_entry = {
			type = je_francia_reuni_bgy_occ
		}
		add_journal_entry = {
			type = je_francia_reuni_bgy_ams
		}
		add_journal_entry = {
			type = je_francia_reuni_bgy_rhi
		}
		add_journal_entry = {
			type = je_francia_reuni_bgy_ibe
		}
		add_loyalists = {
			strata = middle
			value = 0.1
		}
		add_radicals = {
			strata = poor
			value = 0.05
		}
		every_interest_group = {
			limit = {
				OR = {
					is_interest_group_type = ig_armed_forces
				}
			}
			add_modifier = {
				name = modifier_imperialist
				months = 61
				is_decaying = yes
			}
		}
		random_interest_group = {
			limit
			{
				is_interest_group_type = ig_armed_forces
				exist = leader
			}
			leader = {
				save_scope_as = armed_forces_leader
			}
		}
		scope:armed_forces_leader = {
			set_ideology = ideology:ideology_authoritarian
		}
	}
}
